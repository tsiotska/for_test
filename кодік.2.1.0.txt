  function getElementByXpath(path) {
    return document.evaluate(path, document, null, XPathResult.ANY_TYPE, null);
  }

  function findRadio(path) {
    console.log('search radio')
    const result = getElementByXpath(path)
    const first = result.iterateNext()
    console.log(first);
    const second = result.iterateNext()
    console.log(second);
    const third = result.iterateNext()
    console.log(third);
    const fourth = result.iterateNext()
    console.log(fourth);

    const actualDate = fourth ?? third

    if (actualDate) {
      console.log('dispatch')
      actualDate.click()
      setTimeout(() => {
        findButton("//*[@id='btnConfirm']")
      })
    }
  }

  function findButton(path) {
    const result = getElementByXpath(path)
    const first = result.iterateNext()
    first.click()
  }


  document.addEventListener('keydown', function (ev) {
    if (ev.which === 32) {
      findRadio("//input[@type='radio']")
    }
  });
